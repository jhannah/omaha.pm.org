ADOConnection
	$conn=ADOConnection->new();
	$conn->setDB('database name');
	$conn->closeConnection();
	$conn->openConnection('DSN'->'', 'UID'->'', 'PWD'->'', ['database'->'']);
	$conn->begin();
	$conn->commit();
	$conn->rollback();
	$conn->connectionTimeOut(#);
	$conn->commandTimeOut(#);
	$cmd=$conn->newCommand(['command text']);		Returns ADOCommand Ojbect

ADOCommand
	$cmd->setText('command text');
	$cmd->setConnection($conn);
	$cmd->commandTimeOut(#);
	$rs=$cmd->execute();					Returns ADOResultset Object
	$cmd->createParameter('name'->'',
		'type'->$adEmpty
		$adTinyInt $adSmallInt $adInteger $adBigInt
		$adUnsignedTinyInt $adUnsignedSmallInt $adUnsignedInt $adUnsignedBigInt
		$adSingle $adDouble $adCurrency $adDecimal $adNumeric $adBoolean
		$adError $adUserDefined $adVariant $adIDispatch $adIUnknown $adGUID
		$adDate $adDBDate $adDBTime $adDBTimeStamp
		$adBSTR $adChar $adVarChar $adLongVarChar $adWChar $adVarWChar
		$adLongVarWChar $adBinary $adVarBinary $adLongVarBinary,
		'direction'->$adParamUnknown $adParamInput $adParamOutput $adParamInputOutput $adParamReturnValue,
		['value'->'']);
	$cmd->getParameter('name');			returns scalar with value specified by name
	$cmd->setParameter('name'=>'', 'value'=>'');

ADOResultset
	$rs->field('');					returns scalar with value specified by name
	%fields=$rs->fields();				returns hash of fields for current record only
	$rs->setConnection($conn);			Requires a connection object
	$rs->addNew();
	$rs->update();
	$rs->delete();
	$rs->move([#]);						default is move forward one
	$rs->moreResults();				returns boolean indicating weather or not there were more results
	$rs->eof();						boolean true for end of result set
	$rs->bof();						boolean true for beginning of result set that means that you are before first record
	$rs->closeResultset();
	$rs->source('resultset source');
	$rs->open(['resultset source']);
