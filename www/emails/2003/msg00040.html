A drunken OO Perl ode to Monger meetings I wrote a couple months ago:

use Omaha;   # (PerlMongers/Omaha.pm)
use OldChicago;

my $we = new Omaha;
@members = $we->members;
$beers = $we->beers;
my $oc = new OldChicago;

while ($we->conscious and $oc->beerscnt) {
   foreach $pm (@members) {
      if ($beers->empty) {
         $beers->discard;
         $pm->purchase && $oc->beerscnt--;
      }
      $pm->consume($beers);
      $pm->belch;
      $we->hazzah;   #  -grin-
   }
}
